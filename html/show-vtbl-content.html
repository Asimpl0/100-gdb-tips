显示C++ vtable以及相应的内容

例子

#include <iostream>

    struct Base {
      virtual void f(){
        std::cout << "base\n";
      }
    };

    struct Derived : Base {
      void f() override // 'override' is optional
      {
        std::cout << "derived\n";
      }
    };

    int main() {
      Base b;
      Derived d;

      // virtual function call through reference
      Base& br = b; // the type of br is Base&
      Base& dr = d; // the type of dr is Base& as well
    }
